C++14 : Generic Lambda Functions


#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;
int main()
{	
	  // Generic Lambda Display
	auto display = [](auto x){std::cout << x << " ";};

	  // Vector Decleration
	vector<int> a = {1, 2, 3, 4, 5};
	vector<float> b = {1.1f, 2.2f, 3.3f, 4.4f, 5.5f};
	vector<const char*> c = {"The", "Cake", "Is", "A", "Lie"};

	  // Display a <int>
	for_each(begin(a), end(a), display);
	cout << endl;

	  // Display b <float>
	for_each(begin(b), end(b), display);
	cout << endl;

	  // Sort c by the first letter of each word
	sort(begin(c), end(c), [](auto x, auto y){if(x[0] < y[0])return true;return false;});
	  // Display c <const chat*>
	for_each(begin(c), end(c), display);
	cout << endl;

	return 0;
}
