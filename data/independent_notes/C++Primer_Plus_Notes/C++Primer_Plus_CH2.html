C++ Initiation




// myfirst.cpp -- displays a message

#include <iostream>                         // a PREPROCESSOR directive
int main()                                  // function header
{                                           // start of function body
  using namespace std;                      // make definitions visible
  cout << "Come up and C++ me some time.";  // message
  cout << endl;                             // start a new line
  cout << "You won't regret it!" << endl;   // more output
  return 0;                                 // terminate main()
}                                           // end of function body

2. Write a C++ program that asks for a distance in furlongs and converts it to yards.
    (One furlong is 220 yards.)


Answer:


#include <iostream>
#include <stdlib.h>

int FtoY(int furlongs)
{
  return (furlongs * 220);
}

int main(int argc, char* argv[])        // Using main decleration for console input
{
  using namespace std;                  // Setting std to default namespace
  if(argc > 1)                          // If a furlong ammount was provided
    cout << atoi(argv[1])               // Print Output:
         << " Furlongs is equal to "    //  V        V
         << FtoY(atoi(argv[1]))         //  V        V
         << " Yards."                   //  V        V
         << endl;                       //  V        V
  return 0;                             // return 0
}
